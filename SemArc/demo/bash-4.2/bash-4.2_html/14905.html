21     #include <config.h>
22     
23     #if !defined (HAVE_WCSDUP) && defined (HANDLE_MULTIBYTE)
24     
25     #include <stdc.h>
26     #include <wchar.h>
27     #include <bashansi.h>
28     #include <xmalloc.h>
29     
30     wchar_t *
31     wcsdup (ws)
32          const wchar_t *ws;
33     {
34       wchar_t *ret;
35       size_t len;
36     
37       len = wcslen (ws);
38       ret = xmalloc ((len + 1) * sizeof (wchar_t));
39       if (ret == 0)
40         return ret;
41       
42       return (wcscpy (ret, ws));
43     }
44     #endif /* !HAVE_WCSDUP && HANDLE_MULTIBYTE */
