200     int
201     rl_add_funmap_entry (name, function)
202          const char *name;
203          rl_command_func_t *function;
204     {
205       if (funmap_entry + 2 >= funmap_size)
206         {
207           funmap_size += 64;
208           funmap = (FUNMAP **)xrealloc (funmap, funmap_size * sizeof (FUNMAP *));
209         }
210       
211       funmap[funmap_entry] = (FUNMAP *)xmalloc (sizeof (FUNMAP));
212       funmap[funmap_entry]->name = name;
213       funmap[funmap_entry]->function = function;
214     
215       funmap[++funmap_entry] = (FUNMAP *)NULL;
216       return funmap_entry;
217     }
218     
219     static int funmap_initialized;
220     
221     /* Make the funmap contain all of the default entries. */
222     void
223     rl_initialize_funmap ()
224     {
225       register int i;
226     
227       if (funmap_initialized)
228         return;
229     
230       for (i = 0; default_funmap[i].name; i++)
231         rl_add_funmap_entry (default_funmap[i].name, default_funmap[i].function);
232     
233       funmap_initialized = 1;
234       funmap_program_specific_entry_start = i;
235     }

237     /* Produce a NULL terminated array of known function names.  The array
238        is sorted.  The array itself is allocated, but not the strings inside.
239        You should free () the array when you done, but not the pointrs. */
240     const char **
241     rl_funmap_names ()
242     {
243       const char **result;
244       int result_size, result_index;
245     
246       /* Make sure that the function map has been initialized. */
247       rl_initialize_funmap ();
248     
249       for (result_index = result_size = 0, result = (const char **)NULL; funmap[result_index]; result_index++)
250         {
251           if (result_index + 2 > result_size)
252     	{
253     	  result_size += 20;
254     	  result = (const char **)xrealloc (result, result_size * sizeof (char *));
255     	}
256     
257           result[result_index] = funmap[result_index]->name;
258           result[result_index + 1] = (char *)NULL;
259         }
260     
261       qsort (result, result_index, sizeof (char *), (QSFUNC *)_rl_qsort_string_compare);
262       return (result);
263     }
