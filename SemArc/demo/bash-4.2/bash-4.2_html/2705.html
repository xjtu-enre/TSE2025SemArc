28     typedef intmax_t	arrayind_t;
29     
30     enum atype {array_indexed, array_assoc};
31     
32     typedef struct array {
33     	enum atype	type;
34     	arrayind_t	max_index;
35     	int		num_elements;
36     	struct array_element *head;
37     } ARRAY;
38     
39     typedef struct array_element {
40     	arrayind_t	ind;
41     	char	*value;
42     	struct array_element *next, *prev;
43     } ARRAY_ELEMENT;
44     
45     typedef int sh_ae_map_func_t __P((ARRAY_ELEMENT *, void *));
46
